## 1. 性能优化概述

### 1.1 性能目标

| 指标 | 目标值 | 说明 |
|------|--------|------|
| **启动时间** | ≤ 2s | 冷启动 |
| **首屏时间** | ≤ 1s | 首页加载完成 |
| **页面切换** | ≤ 16ms | 60fps |
| **列表滚动** | ≥ 60fps | 流畅无卡顿 |
| **内存占用** | ≤ 200MB | 稳定后 |
| **APK 大小** | ≤ 50MB | 首次安装 |
| **网络请求** | ≤ 1s | 常规接口 |

#### 使用 ListView.builder

```dart

class ImagePreloadManager {
  static final ImagePreloadManager _instance = ImagePreloadManager._internal();
  factory ImagePreloadManager() => _instance;
  ImagePreloadManager._internal();

  final Set<String> _preloadedUrls = {};

  /// 预加载图片
  Future<void> preloadImages(BuildContext context, List<String> urls) async {
    final futures = urls
        .where((url) => !_preloadedUrls.contains(url))
        .map((url) => precacheImage(NetworkImage(url), context))
        .toList();

    await Future.wait(futures);
    _preloadedUrls.addAll(urls);
  }

  void clearPreload() {
    _preloadedUrls.clear();
  }
}
```

## 3. 内存优化

### 3.1 对象池

```dart
/// 对象池 - 复用对象减少 GC
class ObjectPool<T> {
  final Queue<T> _pool = Queue();
  final T Function() _creator;
  final void Function(T)? _reset;
  final int maxSize;

  ObjectPool({
    required T Function() creator,
    void Function(T)? reset,
    this.maxSize = 100,
  })  : _creator = creator,
        _reset = reset;

  /// 获取对象
  T acquire() {
    if (_pool.isEmpty) {
      return _creator();
    }
    return _pool.removeFirst();
  }

  /// 归还对象
  void release(T object) {
    if (_pool.length >= maxSize) {
      return; // 池满了，直接丢弃
    }

    _reset?.call(object);
    _pool.add(object);
  }

  /// 清空对象池
  void clear() {
    _pool.clear();
  }
}

// 使用示例
final listPool = ObjectPool<List<String>>(
  creator: () => <String>[],
  reset: (list) => list.clear(),
);

final list = listPool.acquire();
// 使用 list
listPool.release(list);
```

## 4. 网络优化

### 4.1 请求合并

```dart
/// 批量请求管理器
class BatchRequestManager {
  final Map<String, Future> _pendingRequests = {};

  /// 批量请求 - 相同请求只发一次
  Future<T> batch<T>(
    String key,
    Future<T> Function() request,
  ) async {
    // 如果已有相同请求在执行，等待其完成
    if (_pendingRequests.containsKey(key)) {
      return await _pendingRequests[key] as Future<T>;
    }

    // 发起新请求
    final future = request();
    _pendingRequests[key] = future;

    try {
      final result = await future;
      return result;
    } finally {
      _pendingRequests.remove(key);
    }
  }
}
```

### 4.2 并发控制

```dart
/// 并发请求控制
class ConcurrentRequestManager {
  final int maxConcurrency;

  ConcurrentRequestManager({this.maxConcurrency = 5});

  Future<List<T>> batchConcurrent<T>(
    List<Future<T> Function()> requests,
  ) async {
    final results = <T>[];
    final queue = Queue<Future<T> Function()>.from(requests);
    final futures = <Future<T>>[];

    while (queue.isNotEmpty || futures.isNotEmpty) {
      // 填充并发槽位
      while (futures.length < maxConcurrency && queue.isNotEmpty) {
        futures.add(queue.removeFirst()());
      }

      // 等待任一完成
      if (futures.isNotEmpty) {
        final completed = await Future.any(futures);
        results.add(completed);
        futures.removeWhere((f) => f == completed as Future<T>);
      }
    }

    return results;
  }
}
```

### 4.3 请求缓存

```dart
/// 请求缓存
class RequestCache {
  final Map<String, CacheEntry> _cache = {};
  final Duration defaultDuration;

  RequestCache({this.defaultDuration = const Duration(minutes: 5)});

  Future<T> get<T>(
    String key,
    Future<T> Function() fetcher, {
    Duration? duration,
  }) async {
    // 检查缓存
    final entry = _cache[key];
    if (entry != null && !entry.isExpired) {
      return entry.data as T;
    }

    // 获取数据
    final data = await fetcher();

    // 缓存数据
    _cache[key] = CacheEntry(
      data: data,
      duration: duration ?? defaultDuration,
    );

    return data;
  }

  void clear() {
    _cache.clear();
  }
}

class CacheEntry {
  final dynamic data;
  final DateTime createdAt;
  final Duration duration;

  CacheEntry({
    required this.data,
    required this.duration,
  }) : createdAt = DateTime.now();

  bool get isExpired =>
      DateTime.now().isAfter(createdAt.add(duration));
}
```

## 6. 计算优化

### 6.1 Isolate 使用

```dart
/// Isolate 管理器
class IsolateManager {
  static final IsolateManager _instance = IsolateManager._internal();
  factory IsolateManager() => _instance;
  IsolateManager._internal();

  final Map<String, Isolate> _isolates = {};

  /// 在隔离线程中执行计算
  Future<R> compute<R, T>(
    Future<R> Function(T message) function,
    T message,
  ) async {
    return await dartCompute(function, message);
  }

  /// 创建常驻隔离线程
  Future<void> createPersistentIsolate(
    String name,
    void Function(SendPort) entryPoint,
  ) async {
    final receivePort = ReceivePort();
    final isolate = await Isolate.spawn(
      entryPoint,
      receivePort.sendPort,
    );

    _isolates[name] = isolate;
  }

  /// 关闭隔离线程
  void dispose(String name) {
    _isolates[name]?.kill(priority: Isolate.immediate);
    _isolates.remove(name);
  }
}
```

### 6.2 懒加载

```dart
/// 懒加载列表
class LazyList<T> {
  final List<T> _items = [];
  final Future<List<T>> Function(int page) _loader;
  int _currentPage = 0;
  bool _isLoading = false;
  bool _hasMore = true;

  LazyList(this._loader);

  Future<List<T>> loadMore() async {
    if (_isLoading || !_hasMore) return _items;

    _isLoading = true;
    _currentPage++;

    try {
      final newItems = await _loader(_currentPage);
      _items.addAll(newItems);
      _hasMore = newItems.isNotEmpty;
      return _items;
    } finally {
      _isLoading = false;
    }
  }

  void reset() {
    _items.clear();
    _currentPage = 0;
    _hasMore = true;
    _isLoading = false;
  }
}
```

```dart
class LazyRouter {
  static Route<T> _materialRoute<T>({
    required WidgetBuilder builder,
    RouteSettings? settings,
  }) {
    return MaterialPageRoute<T>(
      builder: builder,
      settings: settings,
    );
  }

  static Route<T> getRoute<T>(RouteSettings settings) {
    switch (settings.name) {
      case '/user/profile':
        return _materialRoute(
          builder: (_) => FutureBuilder(
            future: Future.delayed(const Duration(milliseconds: 100)),
            builder: (_, snapshot) {
              if (snapshot.connectionState == ConnectionState.done) {
                // 懒加载模块
                return UserProfilePage();
              }
              return const Center(child: CircularProgressIndicator());
            },
          ),
          settings: settings,
        );
      default:
        return _materialRoute(
          builder: (_) => const NotFoundPage(),
          settings: settings,
        );
    }
  }
}
```

## 8. 性能监控

### 8.1 性能监控器

```dart
/// 性能监控器
class PerformanceMonitor {
  static final PerformanceMonitor _instance = PerformanceMonitor._internal();
  factory PerformanceMonitor() => _instance;
  PerformanceMonitor._internal();

  final Map<String, Duration> _metrics = {};

  /// 监控函数执行时间
  T track<T>(String name, T Function() func) {
    final stopwatch = Stopwatch()..start();
    try {
      return func();
    } finally {
      stopwatch.stop();
      _metrics[name] = stopwatch.elapsed;

      // 超过阈值则警告
      if (stopwatch.elapsed.inMilliseconds > 100) {
        AppLogger.w('⚠️ Performance Warning: $name took ${stopwatch.elapsed.inMilliseconds}ms');
      }
    }
  }

  /// 监控异步函数执行时间
  Future<T> trackAsync<T>(String name, Future<T> Function() func) async {
    final stopwatch = Stopwatch()..start();
    try {
      return await func();
    } finally {
      stopwatch.stop();
      _metrics[name] = stopwatch.elapsed;
    }
  }

  /// 获取性能报告
  Map<String, Duration> getReport() => Map.from(_metrics);

  /// 清除统计数据
  void clear() {
    _metrics.clear();
  }
}
```

### 8.2 使用示例

